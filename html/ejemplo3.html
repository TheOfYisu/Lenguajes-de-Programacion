// Programa que lee un año y muestra si es o no bisiesto
#include

int bisiesto(int); //declaración o prototipo de la función
int main()
{
int anio;

else

int bisiesto(int a) //definición de la función
{
if(a%4==0 and a%100!=0 or a%400==0)
return 1;
else

}
